\tikzset{
    % --- Node Styles ---
    base box/.style = {
        draw=UofSCDarkGarnet,
        line width=1pt,
        minimum width=3cm,
        minimum height=1.5cm,
        align=center,
        font=\bfseries,
        text=UofSCBlack
    },
    std box/.style = {
        base box,
        fill=UofSCSandstorm
    },
    opt box/.style = {
        base box,
        dashed,
        fill=UofSC10Black
    },
    % --- Connector Styles ---
    % Ethernet: garnet outlines with Atlantic labels
    conn eth/.style = {
        Latex-Latex,
        very thick,
        draw=UofSCAtlantic
    },
    eth label/.style = {
        font=\small\ttfamily,
        midway,
        above,
        text=UofSCAtlantic
    },
    % Power: garnet lines and labels
    conn pwr/.style = {
        -Latex,
        very thick,
        draw=UofSCGarnet
    },
    pwr label/.style = {
        font=\small\ttfamily,
        midway,
        below,
        text=UofSCGarnet
    },
    % --- Layout Configuration ---
    node distance = 2.5cm and 2.5cm
}

\begin{tikzpicture}

    % ==========================================
    % GROUP 1: Legacy Configuration
    % ==========================================
    \begin{scope}[local bounding box=LegacyGroup]
        
        % 1. Main Component Nodes
        \node[std box] (orinA) {Orin PC};
        \node[std box, right=of orinA] (joyA) {Joystick Box};
        \node[std box, right=of joyA] (pwrbox) {Power Box};
        \node[std box, right=of pwrbox] (flirA) {FLIR Camera};

        % 2. Ethernet Connections 
        \draw[conn eth] (orinA)  -- node[eth label] {eth} (joyA);
        \draw[conn eth] (joyA)   -- node[eth label] {eth} (pwrbox);
        \draw[conn eth] (pwrbox) -- node[eth label] {eth} (flirA);

        % 3. Power Inputs
        \coordinate (powerRailA) at ($(orinA.west) + (-2cm, 0)$);

        \node[anchor=east, text=UofSC90Black] (v120A) at (powerRailA |- orinA.west) {120V};
        \node[anchor=east, below=1cm of v120A, text=UofSC90Black] (v5A) {5V};
        \node[anchor=east, below=1cm of v5A, text=UofSC90Black]   (v12A) {12V};

        % 4. Power Wiring
        \draw[conn pwr] (v120A) -- (orinA);
        \draw[conn pwr] (v5A)   -| (joyA.south);
        \draw[conn pwr] (v12A)  -| (pwrbox.south);

        % Special Power Link between Box and Camera
        \draw[conn pwr]
