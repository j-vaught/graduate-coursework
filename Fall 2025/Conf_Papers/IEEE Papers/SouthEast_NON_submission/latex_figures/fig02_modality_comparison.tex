% Fig. 02: Modality comparison for maritime sensing.
% Requires \usepackage{tikz,xcolor} and \definecolor{Garnet}{HTML}{73000A}
\begin{tikzpicture}[x=1cm,y=1cm,line cap=butt,line join=miter,font=\small,text=black]

  \def\panelW{3.4}
  \def\panelH{4.2}
  \def\gap{0.4}
  \pgfmathsetmacro\panelShiftA{0}
  \pgfmathsetmacro\panelShiftB{\panelW+\gap}
  \pgfmathsetmacro\panelShiftC{2*(\panelW+\gap)}

  % --- Panels background ---
  \foreach \xshift in {\panelShiftA,\panelShiftB,\panelShiftC} {
    \filldraw[fill=black!5,draw=black,line width=0.4pt]
      (\xshift,0) rectangle ++(\panelW,\panelH);
  }

  % --- Panel labels ---
  \node[anchor=west] at (0.2, \panelH-0.3) {Marine Radar (Night, Rain, Fog)};
  \node[anchor=west] at (\panelShiftB+0.2, \panelH-0.3) {Camera (Clear Daylight Only)};
  \node[anchor=west] at (\panelShiftC+0.2, \panelH-0.3) {LiDAR (Clear Line-of-Sight)};

  % --- Radar depiction ---
  \foreach \r/\opacity in {1.2/0.8,1.8/0.6,2.4/0.4} {
    \draw[Garnet, line width=0.9pt, opacity=\opacity]
      (0.9,2.2) arc[start angle=180,end angle=360,radius=\r];
  }
  \filldraw[fill=black!20,draw=black] (0.9,2.2) circle [radius=0.2];
  \foreach \dx in {0.3,0.7,1.2} {
    \filldraw[fill=black,draw=black,line width=0.4pt] (0.9+\dx,1.8+0.08*\dx) circle [radius=0.08];
  }
  \node[anchor=west] at (0.2,1.1) {Detects echoes through rain and fog};
  \node[anchor=west] at (0.2,0.8) {Long range, clutter resistant};

  % --- Camera depiction ---
  \filldraw[fill=black!10,draw=black,line width=0.5pt]
    (\panelShiftB+0.3,0.8) rectangle ++(2.0,1.4);
  \draw[black,line width=0.5pt]
    (\panelShiftB+0.5,1.1) -- (\panelShiftB+0.5,1.3);
  \draw[black,line width=0.5pt]
    (\panelShiftB+2.0,1.1) -- (\panelShiftB+2.0,1.3);
  \fill[black!20]
    (\panelShiftB+1.1,1.5) circle [radius=0.3];
  \node[anchor=west] at (\panelShiftB+0.4,0.6) {Strong contrast in daylight only};
  \node[anchor=west] at (\panelShiftB+0.4,0.3) {Fails when visibility drops};

  % --- LiDAR depiction ---
  \foreach \x in {0,0.3,...,1.5} {
    \foreach \y in {0,0.3,...,1.2} {
      \fill[black!70] (\panelShiftC+0.5+\x,0.9+\y) rectangle ++(0.15,0.15);
    }
  }
  \fill[black!10] (\panelShiftC+1.8,2.1) circle [radius=0.25];
  \node[anchor=west] at (\panelShiftC+0.4,0.5) {High detail but weather sensitive};
  \node[anchor=west] at (\panelShiftC+0.4,0.2) {Occlusions break coverage};

  % --- Divider lines ---
  \draw[black,line width=0.6pt] (\panelShiftB,0) -- (\panelShiftB,\panelH);
  \draw[black,line width=0.6pt] (\panelShiftC,0) -- (\panelShiftC,\panelH);

  % --- Bottom annotation ---
  \pgfmathsetmacro\bottomX{1.8+(\panelW+\gap)}
  \node[anchor=center,font=\footnotesize] at (\bottomX,-0.4) {Sensing modalities for maritime object detection};

\end{tikzpicture}
