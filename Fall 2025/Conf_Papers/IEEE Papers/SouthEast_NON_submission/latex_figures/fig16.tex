% Style Guide Colors
\definecolor{Garnet}{HTML}{73000A}

\begin{document}

\begin{tikzpicture}[
    % Global Styles
    node distance=1.5cm and 1.0cm,
    font=\small,
    % Styles for different component types
    hardware/.style={
        rectangle, 
        draw=black, 
        fill=white, 
        line width=0.6pt,
        minimum height=1.2cm,

    process/.style={

    },
    data/.style={

        trapezium right angle=110, 
        draw=black, 
        fill=white, 
        line width=0.6pt,
        minimum height=1.0cm, 
        minimum width=1cm, 
        align=center
    },
    ml/.style={
        rectangle, 
        draw=black, 
        fill=Garnet!20, % Light garnet tint for emphasis

    deployment/.style={

        minimum width=1cm,

    \node[process, right=of csv] (png) {Format Conversion \ \footnotesize CSV $\to$ PNG};

    \node[deployment, left=of train] (deploy) {Output \& \\ Deployment};
